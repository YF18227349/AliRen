config:
    "easysass.compileAfterSave": true,
    "easysass.excludeRegex": "",
    "easysass.formats": [
        {
            "format": "expanded",
            "extension": ".css"
        },
        {
            "format": "compressed",
            "extension": ".min.css"
        }
    ],
